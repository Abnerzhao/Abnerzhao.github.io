字典排序

字典key排序

方法一：

dict = {'b':3,'a':1,'d':7,'c':5}
items = dict.items()
items.sort()

print [value for key,value in items] //[1, 3, 5, 7]
或
print [(k,v) for k,v in items] //[('a', 1), ('b', 3), ('c', 5), ('d', 7)]

方法二：

dict = {'b':3,'a':1,'d':7,'c':5}
print [(k,dict[k]) for k in sorted(dict.keys())] 

//stdout
[('a', 1), ('b', 3), ('c', 5), ('d', 7)]

字典value排序

方法一：

dict = {'b':3,'a':1,'d':7,'c':5}
print [ v for v in sorted(dict.values())]

//stdout
[1, 3, 5, 7]

方法二：

dict = {'b':3,'a':1,'d':7,'c':5}
sorted(dict.items(), lambda x, y: cmp(x[1], y[1]))
//[('a', 1), ('b', 3), ('c', 5), ('d', 7)]
sorted(dict.items(), lambda x, y: cmp(x[1], y[1]), reverse=True) //降序
//[('d', 7), ('c', 5), ('b', 3), ('a', 1)]

列表去重

方法一：

set方法：类似dict但不存储value，且key不能重复

num_list = [1,1,2,3,4,4]
list = list(set(num_list))
print list

//stdout
[1, 2, 3, 4]

方法二：

num_list = [1,1,2,3,4,4]
print {}.fromkeys(num_list).keys() //创建新的字典 list的值为key 然后取key

//stdout
[1, 2, 3, 4]

方法三：

ids = [1,1,2,3,4,4]
func = lambda x,y: x if y in x else x + [y]
print reduce(func, [[], ] + ids) //利用reduce函数

//stdout
[1, 2, 3, 4]

sort和sorted

sort在容器内部排序，sorted生成一个新排好序的容器

sorted(iterable[, cmp[, key[, reverse]]])
iterable:可迭代对象
cmp:比较函数，默认为None
key:比较的关键字，默认为None
reverse:布尔值

利用sort和sorted实现列表排序

#sort
list = [1,3,5,2,7,4]
list.sort(reverse = True)
print list

//stdout
[7, 5, 4, 3, 2, 1]  

#sorted
list = [1,3,5,2,7,4]
new_list = sorted(list)
print list
print new_list

//stdout
[1, 3, 5, 2, 7, 4]
[1, 2, 3, 4, 5, 7]

sorted对字典排序

dict = {'b':3,'a':1,'d':7,'c':5}
print sorted(dict.items(), key=lambda d: d[0]) //key
print sorted(dict.items(), key=lambda d: d[1]) //value

sorted key示例

#exp1
>>> sorted("This is a test string from Andrew".split(), key=str.lower)
['a', 'Andrew', 'from', 'is', 'string', 'test', 'This']

#exp2
>>> student_tuples = [
    ('john', 'A', 15),
    ('jane', 'B', 12),
    ('dave', 'B', 10),
]
>>> sorted(student_tuples, key=lambda student: student[2]) # sort by age
[('dave', 'B', 10), ('jane', 'B', 12), ('john', 'A', 15)]
